class PowerSenseor
  def initialize(i2c_bus=1, dev_addr=0x40)
    @i2c = File.open(sprintf('/dev/i2c-%d', i2c_bus), 'rb+')
    @i2c.ioctl(IOCTL_I2C_SLAVE, dev_addr)

    @c_val = 0

    # shunt resistor = 0.002Î© = 0.00512/(0.002*0.001)=2560 = 0x0a 0x00

  end
