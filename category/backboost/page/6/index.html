/******************************************************

 */

  // correct an error in the resistance value.
  const int CORRECT = 10030;

  int mv = -1;
  int i;
  int adc = 0;
  
  for (i = 0; i<10; i++ ){
    adc = 0;

  }

  mv = mv / 10;

  return mv;
}
